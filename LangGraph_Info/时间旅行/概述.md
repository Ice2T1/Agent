# LangGraph 时间旅行

## 概述

时间旅行功能允许您从先前的检查点恢复执行，支持：

- 🤔 **理解推理**：分析导致成功结果的步骤
- 🐞 **调试错误**：识别错误发生的位置和原因
- 🔍 **探索替代方案**：测试不同的路径以发现更好的解决方案

**重要**：恢复过去的执行会在历史中产生一个新的分支。

## 核心概念

### 检查点（Checkpoint）

- 每个超级步骤保存的图状态快照
- 通过 `thread_id` 和 `checkpoint_id` 唯一标识

### 回放（Replay）

- 从特定检查点恢复执行
- 回放该检查点之前的步骤（不重新执行）
- 执行该检查点之后的步骤（创建新分支）

### 分叉（Fork）

- 从历史中的某个点创建新的执行路径
- 原历史保持不变
- 新分支有自己的检查点历史

## 基本用法

### 1. 查看状态历史

```python
from langgraph.graph import StateGraph
from langgraph.checkpoint.memory import InMemorySaver

checkpointer = InMemorySaver()
graph = builder.compile(checkpointer=checkpointer)

config = {"configurable": {"thread_id": "1"}}

# 执行图
graph.invoke(input_data, config)

# 查看完整历史
history = list(graph.get_state_history(config))

for i, snapshot in enumerate(history):
    print(f"\n=== Checkpoint {i} ===")
    print(f"Checkpoint ID: {snapshot.config['configurable']['checkpoint_id']}")
    print(f"Step: {snapshot.metadata['step']}")
    print(f"Values: {snapshot.values}")
    print(f"Next: {snapshot.next}")
    print(f"Source: {snapshot.metadata['source']}")
```

### 2. 从特定检查点恢复

```python
# 获取历史
history = list(graph.get_state_history(config))

# 选择要恢复的检查点（例如，倒数第 2 个）
target_checkpoint = history[1]  # 索引 0 是最新的
checkpoint_id = target_checkpoint.config["configurable"]["checkpoint_id"]

# 从该检查点恢复
config["configurable"]["checkpoint_id"] = checkpoint_id

# 恢复执行（创建新分支）
result = graph.invoke(None, config)
```

### 3. 修改状态后恢复（探索替代方案）

```python
# 获取历史
history = list(graph.get_state_history(config))
target_checkpoint = history[2]
checkpoint_id = target_checkpoint.config["configurable"]["checkpoint_id"]

# 配置恢复到该检查点
config["configurable"]["checkpoint_id"] = checkpoint_id

# 在恢复前修改状态
graph.update_state(
    config,
    {"key": "new_value"}  # 修改状态
)

# 恢复执行（使用修改后的状态，创建新分支）
result = graph.invoke(None, config)
```

## 实际应用场景

### 场景 1：调试错误

```python
# 执行失败
try:
    result = graph.invoke(input_data, config)
except Exception as e:
    print(f"Error: {e}")
    
    # 查看历史
    history = list(graph.get_state_history(config))
    
    # 找到错误发生前的检查点
    for snapshot in history:
        if snapshot.tasks and snapshot.tasks[0].error:
            print(f"Error in task: {snapshot.tasks[0]}")
            print(f"Error at step: {snapshot.metadata['step']}")
            
            # 回滚到错误前的状态
            error_checkpoint_id = snapshot.parent_config["configurable"]["checkpoint_id"]
            config["configurable"]["checkpoint_id"] = error_checkpoint_id
            
            # 检查错误前的状态
            state = graph.get_state(config)
            print("State before error:", state.values)
            
            # 可以修改状态后重试
            graph.update_state(config, {"problematic_key": "fixed_value"})
            
            # 从错误前的状态恢复
            result = graph.invoke(None, config)
            break
```

### 场景 2：比较不同决策路径

```python
def compare_paths():
    config = {"configurable": {"thread_id": "1"}}
    
    # 路径 A：原始执行
    result_a = graph.invoke(input_data, config)
    print("Path A result:", result_a)
    
    # 获取历史
    history = list(graph.get_state_history(config))
    
    # 路径 B：从第一个决策点分叉
    decision_checkpoint = history[2]  # 假设这是决策点
    config["configurable"]["checkpoint_id"] = decision_checkpoint.config["configurable"]["checkpoint_id"]
    
    # 修改状态以做出不同决策
    graph.update_state(config, {"decision": "alternative_choice"})
    
    # 恢复执行（新分支）
    result_b = graph.invoke(None, config)
    print("Path B result:", result_b)
    
    # 路径 C：从另一个点分叉
    config["configurable"]["checkpoint_id"] = history[1].config["configurable"]["checkpoint_id"]
    graph.update_state(config, {"decision": "third_option"})
    result_c = graph.invoke(None, config)
    print("Path C result:", result_c)
    
    return result_a, result_b, result_c
```

### 场景 3：人工审查后回滚

```python
# 执行包含人工审查的工作流
graph = builder.compile(
    checkpointer=checkpointer,
    interrupt_before=["human_review"]
)

config = {"configurable": {"thread_id": "1"}}

# 执行到审查点
result = graph.invoke(input_data, config)

# 人工审查
state = graph.get_state(config)
print("Pending review:", state.values)

# 审查不通过，需要回滚
if not approve(state.values):
    # 回滚到审查前的状态
    history = list(graph.get_state_history(config))
    pre_review_checkpoint = history[1]
    
    config["configurable"]["checkpoint_id"] = pre_review_checkpoint.config["configurable"]["checkpoint_id"]
    
    # 修改状态
    graph.update_state(config, {
        "feedback": "Please revise",
        "revision_needed": True
    })
    
    # 重新执行
    result = graph.invoke(None, config)
```

## 高级功能

### 1. 查看分支历史

```python
# 获取完整历史（包括所有分支）
history = list(graph.get_state_history(config))

# 每个检查点的 parent_config 指向其父检查点
for snapshot in history:
    parent_id = snapshot.parent_config["configurable"]["checkpoint_id"] if snapshot.parent_config else None
    print(f"Checkpoint: {snapshot.config['configurable']['checkpoint_id']}, Parent: {parent_id}")
```

### 2. 可视化执行树

```python
def visualize_execution_tree(config):
    history = list(graph.get_state_history(config))
    
    # 构建树结构
    tree = {}
    for snapshot in history:
        checkpoint_id = snapshot.config["configurable"]["checkpoint_id"]
        parent_id = snapshot.parent_config["configurable"]["checkpoint_id"] if snapshot.parent_config else "root"
        
        if parent_id not in tree:
            tree[parent_id] = []
        tree[parent_id].append(checkpoint_id)
    
    # 打印树
    def print_tree(node, indent=0):
        print("  " * indent + f"Checkpoint: {node[:8]}...")
        if node in tree:
            for child in tree[node]:
                print_tree(child, indent + 1)
    
    print_tree("root")
```

### 3. 跨线程比较

```python
# 线程 A
config_a = {"configurable": {"thread_id": "1"}}
graph.invoke(input_data, config_a)
history_a = list(graph.get_state_history(config_a))

# 线程 B（相同输入）
config_b = {"configurable": {"thread_id": "2"}}
graph.invoke(input_data, config_b)
history_b = list(graph.get_state_history(config_b))

# 比较相同步骤的状态
for i, (snap_a, snap_b) in enumerate(zip(reversed(history_a), reversed(history_b))):
    print(f"\nStep {i}:")
    print(f"Thread A: {snap_a.values}")
    print(f"Thread B: {snap_b.values}")
```

## 完整示例：调试代理错误

```python
from langgraph.graph import StateGraph, MessagesState, START
from langgraph.checkpoint.memory import InMemorySaver
from langchain_core.messages import HumanMessage, AIMessage
from typing import TypedDict, List

class State(TypedDict):
    messages: List
    iteration: int
    tool_calls: list

# 模拟可能失败的代理
def agent_node(state: MessagesState):
    response = llm.invoke(state["messages"])
    return {"messages": [response]}

def tool_node(state: State):
    # 可能失败的工具调用
    if state["iteration"] > 2:
        raise ValueError("Too many iterations!")
    
    tool_call = state["messages"][-1].tool_calls[0]
    result = execute_tool(tool_call)
    return {"messages": [result], "iteration": state["iteration"] + 1}

# 构建图
builder = StateGraph(State)
builder.add_node("agent", agent_node)
builder.add_node("tool", tool_node)
builder.add_edge(START, "agent")
builder.add_edge("agent", "tool")
builder.add_edge("tool", "agent")

# 编译
checkpointer = InMemorySaver()
graph = builder.compile(checkpointer=checkpointer)

# 执行
config = {"configurable": {"thread_id": "1"}}

try:
    result = graph.invoke(
        {"messages": [HumanMessage(content="Process data")], "iteration": 0},
        config
    )
except Exception as e:
    print(f"Error occurred: {e}")
    
    # 时间旅行调试
    history = list(graph.get_state_history(config))
    
    print("\n=== Execution History ===")
    for i, snapshot in enumerate(history):
        print(f"\nCheckpoint {i}:")
        print(f"  Step: {snapshot.metadata['step']}")
        print(f"  Iteration: {snapshot.values.get('iteration')}")
        print(f"  Messages: {len(snapshot.values.get('messages', []))}")
        
        if snapshot.tasks and snapshot.tasks[0].error:
            print(f"  ERROR: {snapshot.tasks[0].error}")
            
            # 回滚到错误前
            print("\n=== Rolling back ===")
            prev_checkpoint = snapshot.parent_config
            config["configurable"]["checkpoint_id"] = prev_checkpoint["configurable"]["checkpoint_id"]
            
            # 检查状态
            state = graph.get_state(config)
            print(f"State before error: {state.values}")
            
            # 修复并恢复
            graph.update_state(config, {"iteration": 0})  # 重置迭代
            print("Resuming with fixed state...")
            
            try:
                result = graph.invoke(None, config)
                print("Success after rollback!")
            except Exception as e2:
                print(f"Still failing: {e2}")
            
            break
```

## 注意事项

1. **分支管理**：每次从历史恢复都会创建新分支，注意管理分支数量
2. **存储开销**：保留完整历史需要存储空间
3. **检查点 ID**：检查点 ID 可能随 LangGraph 版本变化，不要硬编码
4. **并发访问**：多线程访问同一 thread_id 可能导致冲突
