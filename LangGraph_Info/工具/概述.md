# LangGraph 工具（Tools）

## 什么是工具？

工具是一种封装函数及其输入模式的方式，可以传递给支持工具调用的聊天模型。

**使用场景**：
- 与外部系统交互（API、数据库等）
- 这些系统有特定的输入模式
- 模型需要请求使用特定输入执行函数

## 工具调用基础

### 基本原理

工具调用的关键原则：**模型根据输入的关联性决定何时使用工具**。

```python
from langchain_core.tools import tool
from langchain_openai import ChatOpenAI

# 定义工具
@tool
def multiply(a: int, b: int) -> int:
    """Multiply two numbers."""
    return a * b

# 绑定到 LLM
llm_with_tools = llm.bind_tools([multiply])

# 情况 1：与工具无关的输入
result = llm_with_tools.invoke("Hello world!")
# 结果：AIMessage(content="Hello!") - 没有 tool_calls

# 情况 2：与工具相关的输入
result = llm_with_tools.invoke("What is 2 multiplied by 3?")
# 结果：AIMessage 包含 tool_calls 属性

# 访问工具调用
print(result.tool_calls)
# [{'name': 'multiply', 'args': {'a': 2, 'b': 3}, 'id': 'xxx', 'type': 'tool_call'}]
```

### 工具调用结构

```python
{
    'name': 'multiply',           # 工具名称
    'args': {'a': 2, 'b': 3},     # 工具参数
    'id': 'call_xxx',             # 工具调用 ID
    'type': 'tool_call'           # 类型
}
```

## 创建工具

### 方式 1：使用 @tool 装饰器

```python
from langchain_core.tools import tool

@tool
def search(query: str) -> str:
    """Search for information.
    
    Args:

    return search_api(query)

@tool

    """Calculate mathematical expression."""

```

```python
from langchain_core.tools import BaseTool
```

## 执行工具

### 方式 1：手动执行

```python
def execute_tools(tool_calls: list, tools_by_name: dict):
    results = []
    for tool_call in tool_calls:
        tool_name = tool_call["name"]
        tool_args = tool_call["args"]
        tool_call_id = tool_call["id"]
        
        # 执行工具
        result = tools_by_name[tool_name].invoke(tool_args)
        
        # 创建工具消息
        tool_message = {
            "role": "tool",
            "name": tool_name,
            "content": str(result),
            "tool_call_id": tool_call_id
        }
        results.append(tool_message)
    
    return results

# 使用
response = llm_with_tools.invoke(messages)
if response.tool_calls:
    tool_messages = execute_tools(response.tool_calls, tools_by_name)
    messages.append(response)
    messages.extend(tool_messages)
    
    # 继续对话
    next_response = llm_with_tools.invoke(messages)
```

### 方式 2：使用 ToolNode（预构建）

```python
from langgraph.prebuilt import ToolNode

# 创建 ToolNode
tool_node = ToolNode(tools)

# 在图中使用
from langgraph.graph import StateGraph, MessagesState, START

def call_llm(state: MessagesState):
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}

builder = StateGraph(MessagesState)
builder.add_node("llm", call_llm)
builder.add_node("tools", tool_node)
builder.add_edge(START, "llm")
builder.add_edge("llm", "tools")
builder.add_edge("tools", "llm")  # 循环回到 LLM

graph = builder.compile()
```

## 工具在 LangGraph 中的应用

### 基本工具使用

```python
from langgraph.graph import StateGraph, MessagesState, START, END
from langgraph.prebuilt import ToolNode
from langchain_core.tools import tool
from langchain_openai import ChatOpenAI

# 定义工具
@tool
def get_weather(location: str) -> str:
    """Get the weather for a location."""
    return f"Weather in {location}: Sunny, 25°C"

@tool
def get_time(timezone: str) -> str:
    """Get the current time for a timezone."""
    return f"Time in {timezone}: 10:00 AM"

tools = [get_weather, get_time]
tools_by_name = {t.name: t for t in tools}

# LLM 绑定工具
llm = ChatOpenAI(model="gpt-4")
llm_with_tools = llm.bind_tools(tools)

# 构建代理图
def call_llm(state: MessagesState):
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}

def should_continue(state: MessagesState):
    last_message = state["messages"][-1]
    if last_message.tool_calls:
        return "tools"
    return END

builder = StateGraph(MessagesState)
builder.add_node("llm", call_llm)
builder.add_node("tools", ToolNode(tools))
builder.add_edge(START, "llm")
builder.add_conditional_edges("llm", should_continue)
builder.add_edge("tools", "llm")

graph = builder.compile()

# 使用
result = graph.invoke({
    "messages": [{"role": "user", "content": "What's the weather in Paris?"}]
})
print(result["messages"][-1].content)
```

### 工具交接（Handoffs）

工具可以返回 `Command` 对象实现代理间交接：

```python
from typing import Annotated
from langchain_core.tools import tool, InjectedToolCallId
from langgraph.prebuilt import InjectedState
from langgraph.types import Command

@tool
def transfer_to_specialist(
    state: Annotated[MessagesState, InjectedState],
    tool_call_id: Annotated[str, InjectedToolCallId],
) -> Command:
    """Transfer to a specialist agent."""
    tool_message = {
        "role": "tool",
        "content": "Transferred to specialist",
        "name": "transfer_to_specialist",
        "tool_call_id": tool_call_id,
    }
    
    return Command(
        goto="specialist_agent",
        update={"messages": state["messages"] + [tool_message]},
        graph=Command.PARENT,
    )

# 使用交接工具
tools = [transfer_to_specialist, other_tool]
agent = create_react_agent(model, tools)
```

```python
@tool
def safe_database_query(query: str) -> str:
    """Query the database safely."""
    try:
        result = db.execute(query)
        return str(result)
    except Exception as e:
        return f"Error: {str(e)}"

@tool
def validated_api_call(url: str, params: dict = None) -> str:
    """Call API with validation."""
    if not url.startswith("https://"):
        return "Error: Invalid URL scheme"
    
    try:
        response = requests.get(url, params=params, timeout=10)
        response.raise_for_status()
        return response.text

        return "Error: Request timed out"
    except requests.RequestException as e:
        return f"Error: {str(e)}"
```

## 完整示例：客户服务代理

```python
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool
from langgraph.prebuilt import create_react_agent
from langgraph.graph import MessagesState
from typing import Annotated
import json

# 初始化 LLM
llm = ChatOpenAI(model="gpt-4")

# 定义工具
@tool
def lookup_order(order_id: str) -> str:
    """Look up order status by order ID."""
    # 模拟数据库查询
    orders = {
        "12345": {"status": "Shipped", "date": "2024-01-15"},
        "67890": {"status": "Processing", "date": "2024-01-16"}
    }
    order = orders.get(order_id, {"status": "Not found"})
    return json.dumps(order)

@tool
def process_return(order_id: str, reason: str) -> str:
    """Process a return for an order."""
    # 模拟退货处理
    return f"Return initiated for order {order_id}. Reason: {reason}"

@tool
def check_inventory(product_id: str) -> str:
    """Check product inventory."""
    # 模拟库存查询
    inventory = {
        "PROD001": {"quantity": 50, "warehouse": "A"},
        "PROD002": {"quantity": 0, "warehouse": "B"}
    }
    product = inventory.get(product_id, {"quantity": -1})
    return json.dumps(product)

@tool
def update_customer_email(customer_id: str, new_email: str) -> str:
    """Update customer email address."""
    # 模拟更新
    return f"Email updated for customer {customer_id}"

tools = [
    lookup_order,
    process_return,
    check_inventory,
    update_customer_email
]

# 创建代理
agent = create_react_agent(
    llm,
    tools,
    prompt="""You are a customer service assistant.
    
You can help customers with:
- Checking order status
- Processing returns
- Checking product inventory
- Updating contact information

Always be polite and helpful. If you don't have enough information,
ask the customer for clarification before calling tools.

When using tools, explain what you're doing to the customer."""
)

# 使用示例
def customer_service_chat():
    messages = []
    
    while True:
        user_input = input("Customer: ")
        if user_input.lower() in ["quit", "exit"]:
            break
        
        messages.append({"role": "user", "content": user_input})
        
        # 调用代理
        result = agent.invoke({"messages": messages})
        
        # 获取代理响应
        assistant_message = result["messages"][-1].content
        print(f"Assistant: {assistant_message}")
        
        messages.append({"role": "assistant", "content": assistant_message})

# 测试场景
test_scenarios = [
    "Where is my order 12345?",
    "I want to return order 67890 because it's damaged",
    "Do you have PROD001 in stock?",
    "Update my email to new@example.com"
]

for scenario in test_scenarios:
    print(f"\n=== Test: {scenario} ===")
    result = agent.invoke({
        "messages": [{"role": "user", "content": scenario}]
    })
    print(f"Response: {result['messages'][-1].content}")
```

## 故障排除

### Q: 工具不被调用？

检查：
1. 工具描述是否清晰
2. LLM 是否正确绑定工具
3. 输入是否足够触发工具调用

### Q: 工具调用失败？

```python
# 添加详细日志

logging.basicConfig(level=logging.DEBUG)

# 检查工具调用
response = llm_with_tools.invoke(messages)

# 手动执行工具看是否出错
for tool_call in response.tool_calls:

        result = tools_by_name[tool_call["name"]].invoke(tool_call["args"])

    except Exception as e:
```

### Q: 如何限制工具调用次数？

```python
def call_llm_with_limit(state: MessagesState):
    # 计算已进行的工具调用次数
    tool_call_count = sum(
        1 for msg in state["messages"]
        if msg.type == "tool"
    )
    
    if tool_call_count > 5:
        # 超过限制，强制结束
        return {"messages": [AIMessage(content="Maximum tool calls reached")]}
    
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}
```

### Q: 如何处理并行工具调用？

```python
def execute_tools_parallel(tool_calls: list):
    # 并行执行所有工具调用
    with ThreadPoolExecutor() as executor:
        futures = [
            executor.submit(
                tools_by_name[tc["name"]].invoke,
                tc["args"]
            )
            for tc in tool_calls
        ]
        results = [f.result() for f in futures]
    
    return results
```
