# LangGraph 子图（Subgraphs）

## 什么是子图？

**子图**是一个图，它被用作另一个图中的一个节点——这是封装概念在 LangGraph 中的应用。

子图允许构建包含多个组件（这些组件本身也是图）的复杂系统。

## 使用子图的原因

### 1. 构建多代理系统
- 每个代理作为独立子图
- 父图协调代理间通信

### 2. 重用组件
- 在多个图中重用一组节点
- 一次定义，多处使用

### 3. 团队独立开发
- 不同团队独立处理图的不同部分
- 只要子图接口（输入/输出模式）遵守，父图无需了解子图细节

## 子图通信机制

子图作为父图中的一个节点，需要与父图传递状态。有两种情况：

### 情况 1：共享状态键

父图和子图在状态模式中具有**共享状态键**。

```python
from langgraph.graph import StateGraph, MessagesState, START

# 子图和父图共享 MessagesState
class State(MessagesState):
    pass

# === 子图 ===
def call_model(state: MessagesState):
    response = model.invoke(state["messages"])
    return {"messages": response}

subgraph_builder = StateGraph(MessagesState)
subgraph_builder.add_node(call_model)
subgraph_builder.add_edge(START, "call_model")
subgraph = subgraph_builder.compile()

# === 父图 ===
builder = StateGraph(MessagesState)
builder.add_node("subgraph_node", subgraph)  # 子图作为节点
builder.add_edge(START, "subgraph_node")
graph = builder.compile()

# 调用
result = graph.invoke({"messages": [{"role": "user", "content": "hi!"}]})
```

**关键点**：
- 父图和子图共享 `MessagesState`
- 子图直接作为节点添加
- 状态自动传递

### 情况 2：不同状态模式

父图和子图具有**不同的模式**（没有共享状态键）。

```python
from typing_extensions import TypedDict, Annotated
from langchain_core.messages import AnyMessage
from langgraph.graph import StateGraph, MessagesState, START
from langgraph.graph.message import add_messages

# 子图状态
class SubgraphState(TypedDict):
    subgraph_messages: Annotated[list[AnyMessage], add_messages]

# 父图状态
class ParentState(TypedDict):
    messages: Annotated[list[AnyMessage], add_messages]
    extra_data: str

# === 子图 ===
def call_model(state: SubgraphState):
    response = model.invoke(state["subgraph_messages"])
    return {"subgraph_messages": response}

subgraph_builder = StateGraph(SubgraphState)
subgraph_builder.add_node(call_model)
subgraph_builder.add_edge(START, "call_model")
subgraph = subgraph_builder.compile()

# === 父图 ===
def call_subgraph(state: ParentState):
    # 状态转换：父图 -> 子图
    subgraph_input = {"subgraph_messages": state["messages"]}
    
    # 调用子图
    response = subgraph.invoke(subgraph_input)
    
    # 状态转换：子图 -> 父图
    return {"messages": response["subgraph_messages"]}

builder = StateGraph(ParentState)
builder.add_node("subgraph_node", call_subgraph)
builder.add_edge(START, "subgraph_node")
graph = builder.compile()

# 调用
result = graph.invoke({
    "messages": [{"role": "user", "content": "hi!"}],
    "extra_data": "some data"
})
```

**关键点**：
- 父图和子图有不同的状态模式
- 需要在节点内进行状态转换
- 提供更大的灵活性

## 完整示例：多代理系统中的子图

```python
from langchain_openai import ChatOpenAI
from langgraph.graph import StateGraph, MessagesState, START, END
from langgraph.types import Command
from typing import Literal

model = ChatOpenAI()

# === 研究代理（子图）===
class ResearchState(MessagesState):
    query: str
    findings: list

def search_web(state: ResearchState):
    results = web_search(state["query"])
    return {"findings": [results]}

def analyze_results(state: ResearchState):
    analysis = model.invoke([
        {"role": "system", "content": "Analyze search results"},
        *state["messages"],
        {"role": "user", "content": str(state["findings"])}
    ])
    return {"messages": [analysis]}

research_builder = StateGraph(ResearchState)
research_builder.add_node("search", search_web)
research_builder.add_node("analyze", analyze_results)
research_builder.add_edge(START, "search")
research_builder.add_edge("search", "analyze")
research_builder.add_edge("analyze", END)
research_agent = research_builder.compile()

# === 写作代理（子图）===
class WritingState(MessagesState):
    draft: str

def write_draft(state: WritingState):
    draft = model.invoke([
        {"role": "system", "content": "Write a comprehensive draft"},
        *state["messages"]
    ])
    return {"draft": draft.content, "messages": [draft]}

def review_draft(state: WritingState):
    # 简单审查逻辑
    if len(state["draft"]) > 100:
        return {"messages": [AIMessage(content="Draft approved")], "final": state["draft"]}
    else:
        return {"messages": [AIMessage(content="Draft too short")]}

writing_builder = StateGraph(WritingState)
writing_builder.add_node("write", write_draft)
writing_builder.add_node("review", review_draft)
writing_builder.add_edge(START, "write")
writing_builder.add_edge("write", "review")
writing_builder.add_edge("review", END)
writing_agent = writing_builder.compile()

# === 父图（协调器）===
class CoordinatorState(MessagesState):
    topic: str
    research_findings: list
    final_content: str

def coordinate_research(state: CoordinatorState) -> Command[Literal["research_agent"]]:
    # 准备子图输入
    research_input = {
        **state,
        "query": state["topic"],
        "messages": [{"role": "user", "content": f"Research: {state['topic']}"}]
    }
    
    # 调用研究子图
    research_result = research_agent.invoke(research_input)
    
    return {
        "research_findings": research_result["findings"],
        "messages": research_result["messages"]
    }

def coordinate_writing(state: CoordinatorState) -> Command[Literal["writing_agent"]]:
    # 准备子图输入
    writing_input = {
        **state,
        "messages": [
            {"role": "system", "content": "Write based on research"},
            *state["messages"],
            {"role": "user", "content": f"Findings: {state['research_findings']}"}
        ]
    }
    
    # 调用写作子图
    writing_result = writing_agent.invoke(writing_input)
    
    return {
        "final_content": writing_result.get("final", ""),
        "messages": writing_result["messages"]
    }

coordinator_builder = StateGraph(CoordinatorState)
coordinator_builder.add_node("research", coordinate_research)
coordinator_builder.add_node("writing", coordinate_writing)
coordinator_builder.add_edge(START, "research")
coordinator_builder.add_edge("research", "writing")
coordinator_builder.add_edge("writing", END)

graph = coordinator_builder.compile()

# 使用
result = graph.invoke({
    "topic": "Climate change impact on agriculture",
    "messages": []
})

print(f"Final content: {result['final_content']}")
```

## 子图作为可重用组件

```python
# 定义一个通用的 QA 子图
def create_qa_agent(model, tools, prompt):
    class QAState(MessagesState):
        answer: str
    
    def answer_question(state: QAState):
        response = model.invoke([
            {"role": "system", "content": prompt},
            *state["messages"]
        ])
        return {"messages": [response], "answer": response.content}
    
    builder = StateGraph(QAState)
    builder.add_node("answer", answer_question)
    builder.add_edge(START, "answer")
    
    return builder.compile()

# 在不同场景重用
medical_qa = create_qa_agent(
    model=medical_model,
    tools=[medical_database],
    prompt="You are a medical assistant"
)

legal_qa = create_qa_agent(
    model=legal_model,
    tools=[legal_database],
    prompt="You are a legal assistant"
)

technical_qa = create_qa_agent(
    model=tech_model,
    tools=[code_database],
    prompt="You are a technical assistant"
)

# 在父图中使用
class RouterState(MessagesState):
    category: str

def route_question(state: RouterState):
    # 简单路由逻辑
    if "medical" in state["messages"][-1].content:
        return "medical_qa"
    elif "legal" in state["messages"][-1].content:
        return "legal_qa"
    else:
        return "technical_qa"

parent_builder = StateGraph(RouterState)
parent_builder.add_node("medical_qa", medical_qa)
parent_builder.add_node("legal_qa", legal_qa)
parent_builder.add_node("technical_qa", technical_qa)
parent_builder.add_conditional_edges(START, route_question)

router_graph = parent_builder.compile()
```

## 子图与 Command 配合使用

在子图中使用 `Command` 导航到父图：

```python
from langgraph.types import Command

# 子图中的节点
def child_node(state: ChildState) -> Command[Literal["another_child"]]:
    if needs_parent_action(state):
        # 导航到父图中的节点
        return Command(
            goto="parent_action_node",
            update={"data_from_child": "some data"},
            graph=Command.PARENT  # 关键：导航到父图
        )
    
    return Command(goto="another_child")

# 包装子图以便可视化
def call_child(state: ParentState) -> Command[Literal["next_parent_node"]]:
    result = child_graph.invoke(state)
    return Command(
        goto="next_parent_node",
        update=result
    )

parent_builder = StateGraph(ParentState)
parent_builder.add_node("child", call_child)  # 使用包装函数
parent_builder.add_node("parent_action_node", parent_action)
parent_builder.add_edge(START, "child")
```

## 嵌套子图

子图可以包含子图，形成层次结构：

```python
# 层级 3：最深层子图
deep_subgraph = create_deep_subgraph()

middle_builder = StateGraph(MiddleState)
```

## 子图流式传输

```python
# 启用子图流式传输
for chunk in graph.stream(input_data, subgraphs=True):
    if isinstance(chunk, tuple):
        # (namespace, update) 元组
        namespace, update = chunk
        print(f"Subgraph {'.'.join(namespace)}: {update}")
    else:
        # 父图更新
        print(f"Parent: {chunk}")
```

**namespace 示例**：
- `("research_agent",)` - 研究子图
- `("writing_agent",)` - 写作子图
- `("research_agent", "search")` - 研究子图中的 search 节点

```python
# 子图实现

    # 实现
```

### 2. 状态隔离

```python
# 子图使用独立的状态键
class ChildState(TypedDict):
    child_internal_data: str
    child_messages: list

# 父图使用不同的键
class ParentState(TypedDict):
    parent_data: str
    parent_messages: list

# 在边界处进行转换
def call_child(state: ParentState):
    child_input = {
        "child_internal_data": state["parent_data"],
        "child_messages": state["parent_messages"]
    }
    result = child_graph.invoke(child_input)
    return {
        "parent_data": result["child_internal_data"],
        "parent_messages": result["child_messages"]
    }
```

### 3. 错误处理

```python
def safe_subgraph_call(state: ParentState):
    try:
        result = subgraph.invoke(state)
        return {"success": True, **result}
    except Exception as e:
        return {"success": False, "error": str(e)}
```

## 完整示例：客户支持系统

```python
from langchain_openai import ChatOpenAI
from langgraph.graph import StateGraph, MessagesState, START, END
from typing import TypedDict, List

model = ChatOpenAI()

# === 分类子图 ===
class ClassificationState(TypedDict):
    message: str
    category: str
    urgency: str

def classify_category(state: ClassificationState):
    categories = ["technical", "billing", "general"]
    prompt = f"Classify: {state['message']}"
    category = model.invoke(prompt).content
    return {"category": category}

def classify_urgency(state: ClassificationState):
    urgencies = ["low", "medium", "high"]
    prompt = f"Urgency: {state['message']}"
    urgency = model.invoke(prompt).content
    return {"urgency": urgency}

classifier_builder = StateGraph(ClassificationState)
classifier_builder.add_node("category", classify_category)
classifier_builder.add_node("urgency", classify_urgency)
classifier_builder.add_edge(START, "category")
classifier_builder.add_edge("category", "urgency")
classifier_builder.add_edge("urgency", END)
classifier = classifier_builder.compile()

# === 技术支持子图 ===
class TechState(MessagesState):
    solution: str

def search_knowledge_base(state: TechState):
    kb_results = search_kb(state["messages"][-1].content)
    return {"context": kb_results}

def generate_solution(state: TechState):
    solution = model.invoke([
        {"role": "system", "content": "Provide technical solution"},
        {"role": "user", "content": f"Context: {state['context']}\nIssue: {state['messages'][-1].content"}
    ])
    return {"solution": solution.content, "messages": [solution]}

tech_builder = StateGraph(TechState)
tech_builder.add_node("search", search_knowledge_base)
tech_builder.add_node("solve", generate_solution)
tech_builder.add_edge(START, "search")
tech_builder.add_edge("search", "solve")
tech_builder.add_edge("solve", END)
tech_support = tech_builder.compile()

# === 计费支持子图 ===
class BillingState(MessagesState):
    account_info: dict
    response: str

def get_account_info(state: BillingState):
    account = lookup_account(state["messages"][-1].content)
    return {"account_info": account}

def generate_billing_response(state: BillingState):
    response = model.invoke([
        {"role": "system", "content": "Handle billing inquiries"},
        {"role": "user", "content": f"Account: {state['account_info']}\nQuestion: {state['messages'][-1].content"}
    ])
    return {"response": response.content, "messages": [response]}

billing_builder = StateGraph(BillingState)
billing_builder.add_node("lookup", get_account_info)
billing_builder.add_node("respond", generate_billing_response)
billing_builder.add_edge(START, "lookup")
billing_builder.add_edge("lookup", "respond")
billing_builder.add_edge("respond", END)
billing_support = billing_builder.compile()

# === 父图（路由和协调）===
class SupportState(MessagesState):
    category: str
    urgency: str
    final_response: str

def route_to_specialist(state: SupportState):
    category = state["category"]
    
    if category == "technical":
        tech_input = {**state, "messages": state["messages"]}
        result = tech_support.invoke(tech_input)
        return {"final_response": result["solution"], "messages": result["messages"]}
    elif category == "billing":
        billing_input = {**state, "messages": state["messages"]}
        result = billing_support.invoke(billing_input)
        return {"final_response": result["response"], "messages": result["messages"]}
    else:
        # 一般咨询
        response = model.invoke(state["messages"])
        return {"final_response": response.content, "messages": [response]}

support_builder = StateGraph(SupportState)
support_builder.add_node("classify", classifier)
support_builder.add_node("specialist", route_to_specialist)
support_builder.add_edge(START, "classify")
support_builder.add_edge("classify", "specialist")
support_builder.add_edge("specialist", END)

graph = support_builder.compile()

# 使用
result = graph.invoke({
    "messages": [{"role": "user", "content": "My internet is not working"}]
})

print(f"Response: {result['final_response']}")
```

## 故障排除

### Q: 子图状态不更新？

确保：
1. 状态键在父图和子图中正确映射
2. 返回了正确的更新字典
3. 使用了正确的 reducer
