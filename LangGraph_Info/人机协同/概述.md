# LangGraph 人机协同（Human-in-the-Loop）

## 概述

LangGraph 支持强大的人工参与循环（HIL）工作流，允许在自动化过程中的任何环节进行人工干预。这在 LLM 驱动的应用中尤其有用，因为模型输出可能需要验证、更正或额外的上下文。

## 主要功能

### 1. 持久化执行状态
- 在每个步骤后检查图状态
- 允许在定义好的节点处无限期暂停
- 支持异步人工审查或输入，不受时间限制

### 2. 灵活的集成点
- HIL 逻辑可以在工作流的任何点引入
- 支持有针对性的人工参与：
  - 批准 API 调用
  - 更正输出
  - 引导对话

## 典型用例

### 🛠️ 审查工具调用
在工具执行之前，人工可以审查、编辑或批准 LLM 请求的工具调用。

### ✅ 验证 LLM 输出
人工可以审查、编辑或批准 LLM 生成的内容。

### 💡 提供上下文
使 LLM 能够明确请求人工输入以进行澄清或提供额外细节，或支持多轮对话。

## 核心原语

### 1. interrupt() 函数

在特定点暂停执行，并显示信息供人工审查：

```python
from langgraph.types import interrupt

def human_review_node(state: State):
    # 显示需要审查的信息
    print("需要审查:")
    print(f"内容：{state['pending_content']}")
    
    # 暂停并等待人工输入
    user_input = interrupt(value="请审查并提供反馈")
    
    # 处理反馈
    return {"review_result": user_input}
```

### 2. Command 原语

用于使用人工提供的值恢复执行：

```python
from langgraph.types import Command

def resume_node(state: State):
    # 从 interrupt 恢复后继续执行
    feedback = state.get("review_feedback")
    
    if feedback == "approve":
        return Command(goto="next_node")
    else:
        return Command(goto="revise_node", update={"feedback": feedback})
```

## 使用方式

### 方式 1：使用 interrupt() 暂停

```python
from langgraph.graph import StateGraph, MessagesState, START
from langgraph.types import interrupt
from langgraph.checkpoint.memory import InMemorySaver

class State(TypedDict):
    messages: list
    status: str

# 检查点器（必需）
checkpointer = InMemorySaver()

def agent_node(state: MessagesState):
    response = llm.invoke(state["messages"])
    return {"messages": [response]}

def human_node(state: MessagesState):
    # 显示当前状态
    print("=== 需要人工审查 ===")
    print(f"最后消息：{state['messages'][-1].content}")
    
    # 暂停并等待输入
    user_input = interrupt(value="Ready for user input")
    
    # 返回用户输入
    return {"messages": [{"role": "user", "content": user_input}]}

# 构建图
builder = StateGraph(MessagesState)
builder.add_node("agent", agent_node)
builder.add_node("human", human_node)
builder.add_edge(START, "agent")
builder.add_edge("agent", "human")
builder.add_edge("human", "agent")

# 编译（必须有 checkpointer）
graph = builder.compile(checkpointer=checkpointer)

# 使用
config = {"configurable": {"thread_id": "1"}}

# 第一次调用：执行到 human_node 并暂停
result = graph.invoke(
    {"messages": [{"role": "user", "content": "Help me write code"}]},
    config
)

# 此时执行暂停，等待人工输入

# 使用 Command 恢复执行
result = graph.invoke(
    Command(resume="Please add error handling"),
    config
)
```

### 方式 2：使用断点审查

```python
from langgraph.graph import StateGraph

def tool_execution_node(state: State):
    # 执行工具
    tool_call = state["tool_calls"][-1]
    result = tools[tool_call["name"]].invoke(tool_call["args"])

config = {"configurable": {"thread_id": "1"}}

# 执行到断点

state = graph.get_state(config)

    "tool_calls": [{"name": "safe_tool", "args": {...}}]
})

# 恢复执行
result = graph.invoke(None, config)
```

### 方式 3：多轮对话

```python
from typing import Literal

class MultiTurnState(MessagesState):
    last_active_agent: str

def human_node(state: MultiTurnState) -> Command[Literal["agent_a", "agent_b"]]:
    """收集用户输入并路由到活跃的代理"""
    user_input = interrupt(value="Ready for user input")
    
    # 确定哪个代理是活跃的
    active_agent = state["last_active_agent"]
    
    return Command(
        update={
            "messages": [{"role": "user", "content": user_input}]
        },
        goto=active_agent
    )

def agent_a(state: MultiTurnState) -> Command[Literal["agent_b", "human"]]:
    response = agent_a_llm.invoke(state["messages"])
    
    # 决定是继续还是转交
    if needs_specialist(response):
        return Command(
            goto="agent_b",
            update={"messages": [response], "last_active_agent": "agent_a"}
        )
    else:
        return Command(
            goto="human",
            update={"messages": [response], "last_active_agent": "agent_a"}
        )

def agent_b(state: MultiTurnState) -> Command[Literal["agent_a", "human"]]:
    response = agent_b_llm.invoke(state["messages"])
    return Command(
        goto="human",
        update={"messages": [response], "last_active_agent": "agent_b"}
    )

# 构建图
builder = StateGraph(MultiTurnState)
builder.add_node("agent_a", agent_a)
builder.add_node("agent_b", agent_b)
builder.add_node("human", human_node)
builder.add_edge(START, "agent_a")

graph = builder.compile(checkpointer=checkpointer)

# 多轮对话
config = {"configurable": {"thread_id": "1"}}

# 第 1 轮
result = graph.invoke(
    {"messages": [{"role": "user", "content": "I need travel advice"}]},
    config
)

# 第 2 轮（恢复）
result = graph.invoke(
    Command(resume="Can you suggest hotels?"),
    config
)

# 第 3 轮
result = graph.invoke(
    Command(resume="I prefer budget options"),
    config
)
```

## 实际应用场景

### 场景 1：审批工作流

```python
from enum import Enum

class ApprovalStatus(str, Enum):
    APPROVED = "approved"
    REJECTED = "rejected"
    NEEDS_REVISION = "needs_revision"

class ApprovalState(TypedDict):
    document: str
    approval_status: Optional[ApprovalStatus]
    feedback: Optional[str]

def generate_document(state: ApprovalState):
    doc = llm.invoke("Generate report")
    return {"document": doc}

def approval_node(state: ApprovalState):
    print("=== 文档审批 ===")
    print(f"文档内容：{state['document']}")
    
    # 暂停等待审批
    decision = interrupt(
        value="审批 (approve/reject/revise): "
    )
    
    # 如果需要修改，收集反馈
    feedback = None
    if decision == "needs_revision":
        feedback = interrupt(value="请提供修改意见：")
    
    return {
        "approval_status": decision,
        "feedback": feedback
    }

def process_approval(state: ApprovalState):
    status = state["approval_status"]
    
    if status == "approved":
        # 继续后续流程
        return {"final": True}
    elif status == "rejected":
        # 结束或重新开始
        return {"final": False}
    else:
        # 需要修改
        revised_doc = llm.invoke(
            f"Revise based on feedback: {state['feedback']}\n\nOriginal: {state['document']}"
        )
        return {"document": revised_doc, "approval_status": None}

# 构建审批流程
builder = StateGraph(ApprovalState)
builder.add_node("generate", generate_document)
builder.add_node("approve", approval_node)
builder.add_node("revise", process_approval)
builder.add_edge(START, "generate")
builder.add_edge("generate", "approve")
builder.add_conditional_edges(
    "approve",
    lambda s: "revise" if s["approval_status"] == "needs_revision" else "end",
    {"revise": "revise", "end": END}
)
builder.add_edge("revise", "approve")

graph = builder.compile(checkpointer=checkpointer)

# 使用
config = {"configurable": {"thread_id": "1"}}
result = graph.invoke({"document": ""}, config)
```

### 场景 2：代码审查

```python
def code_generation_node(state: State):
    code = llm.invoke("Generate Python function")
    return {"code": code}

def code_review_node(state: State):
    print("=== 代码审查 ===")
    print(state["code"])
    
    # 收集审查意见
    review = interrupt(
        value="审查意见 (pass/fail + 评论): "
    )
    
    return {"review": review}

def apply_review(state: State):
    if "pass" in state["review"].lower():
        return {"approved": True}
    else:
        # 提取评论并修改
        comments = state["review"]
        revised = llm.invoke(
            f"Revise code based on: {comments}\n\nOriginal:\n{state['code']}"
        )
        return {"code": revised, "approved": False}

builder = StateGraph(State)
builder.add_node("generate", code_generation_node)
builder.add_node("review", code_review_node)
builder.add_node("apply", apply_review)
builder.add_edge(START, "generate")
builder.add_edge("generate", "review")
builder.add_edge("review", "apply")
builder.add_conditional_edges(
    "apply",
    lambda s: END if s["approved"] else "review"
)

graph = builder.compile(checkpointer=checkpointer)
```

```python
import time

def timeout_human_node(state: State, config: RunnableConfig):
    # 记录暂停时间
    if "pause_time" not in state:
        return {"pause_time": time.time()}
    
    # 检查是否超时（例如 24 小时）
    elapsed = time.time() - state["pause_time"]
    if elapsed > 24 * 3600:
        print("超时，自动继续")
        return {"timeout": True}
    
    user_input = interrupt(value="请审查：")
    return {"user_input": user_input}
```

## 完整示例：内容审核系统

```python
from langgraph.graph import StateGraph, MessagesState, START
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import InMemorySaver
from typing import TypedDict, List, Optional, Literal

class ContentState(TypedDict):
    content: str
    category: Optional[str]
    flagged: bool
    moderator_decision: Optional[str]
    final_result: str

# 检查点器
checkpointer = InMemorySaver()

# 自动分类
def classify_content(state: ContentState):
    category = llm.invoke(f"Classify: {state['content']}")
    flagged = is_sensitive_category(category)
    
    return {
        "category": category,
        "flagged": flagged
    }

# 人工审核
def moderator_review(state: ContentState):
    print("\n" + "="*60)
    print("内容审核")
    print("="*60)
    print(f"内容：{state['content']}")
    print(f"自动分类：{state['category']}")
    print(f"已标记：{state['flagged']}")
    print("="*60)
    
    decision = interrupt(
        value="决策 (approve/reject/escalate): "
    )
    
    feedback = None
    if decision == "reject":
        feedback = interrupt(value="拒绝原因：")
    
    return {
        "moderator_decision": decision,
        "feedback": feedback
    }

# 处理决策
def process_decision(state: ContentState) -> Command[Literal["escalate", "finalize"]]:
    decision = state["moderator_decision"]
    
    if decision == "approve":
        return Command(
            goto="finalize",
            update={"final_result": "approved"}
        )
    elif decision == "reject":
        return Command(
            goto="finalize",
            update={"final_result": f"rejected: {state['feedback']}"}
        )
    else:  # escalate
        return Command(
            goto="escalate",
            update={"final_result": "escalated"}
        )

# 升级处理
def escalate_review(state: ContentState):
    print("\n升级到高级审核员")
    senior_decision = interrupt(value="高级审核决策：")
    return {"final_result": f"escalated: {senior_decision}"}

# 构建工作流
builder = StateGraph(ContentState)
builder.add_node("classify", classify_content)
builder.add_node("review", moderator_review)
builder.add_node("process", process_decision)
builder.add_node("escalate", escalate_review)
builder.add_node("finalize", lambda s: s)

builder.add_edge(START, "classify")
builder.add_conditional_edges(
    "classify",
    lambda s: "review" if s["flagged"] else "finalize",
    {"review": "review", "finalize": "finalize"}
)
builder.add_edge("review", "process")
builder.add_edge("escalate", "finalize")

graph = builder.compile(checkpointer=checkpointer)

# 使用
config = {"configurable": {"thread_id": "1"}}

# 提交内容审核
result = graph.invoke({
    "content": "User submitted content...",
    "flagged": False
}, config)

# 如果需要人工审核，执行会暂停
# 使用 Command 恢复
result = graph.invoke(
    Command(resume="approve"),
    config
)

print(f"最终结果：{result['final_result']}")
```

## 故障排除

### Q: interrupt() 不工作？

确保：
1. 已设置 `checkpointer`
2. 在正确的执行上下文中调用
3. 使用 `Command(resume=...)` 恢复

### Q: 如何调试人工节点？

```python
# 启用详细日志
import logging
logging.basicConfig(level=logging.DEBUG)

# 检查状态
state = graph.get_state(config)
print("当前状态:", state.values)
print("下一个节点:", state.next)
```

### Q: 可以跳过人工节点吗？

```python
# 可以通过更新状态来"跳过"
graph.update_state(config, {
    "skip_human": True,
    "auto_decision": "approved"
})

# 然后恢复执行
graph.invoke(None, config)
```
